0:             NMT node control
0x080:         SYNC
0x80 - 0x100:  Emergency
100:           Time Stamp
0x180 - 0x200: PDO1 tx
0x200 - 0x280: PDO1 rx
0x280 - 0x300: PDO2 tx
0x300 - 0x380: PDO2 rx
0x380 - 0x400: PDO3 tx
0x400 - 0x480: PDO3 rx
0x480 - 0x500: PDO4 tx
0x500 - 0x580: PDO4 rx
0x580 - 0x600: SDO tx
0x600 - 0x680: SDO rx
0x700 - 0x7FF: Heartbeats
